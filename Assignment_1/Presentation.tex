\subsection{Find Value of k}
\begin{frame}
\frametitle{Find Value of k}
Given that Area of triangle is 4
 
 \begin{equation}
 \begin{align}
 Area of Triangle = \frac12  \times \mydet{Area Matrix}
 \end{align}
 \end{equation}

\begin{equation} 
\label{eq1}
4 = \frac12 \times  \begin{vmatrix}
1 & 1 & 1  \\ 
\vec{A} & \vec{B} & \vec{C} 
\end{vmatrix}     
    \label{eq:myeqn}
\end{equation}

\begin{equation} 
\label{eq1}
4= \frac12 \times  \begin{vmatrix}
1 & 1 & 1  \\ 
k & 4 & 0 \\ 
0 & 0 & 2
\end{vmatrix}     
    \label{eq:myeqn}
\end{equation}
